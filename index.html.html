// api/data.js let items = []; // Временная память export default function handler(req, res) { if (req.method === 'POST') { // Сюда приходят данные от парсера const { price, link } = req.body; items.push({ price, link, time: new Date().toLocaleTimeString() }); return res.status(200).json({ status: "ok" }); } else { // Сюда стучится Mini App, чтобы показать список return res.status(200).json(items.slice(-20)); // Отдаем последние 20 товаров } }
